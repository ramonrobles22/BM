---- CATALOGO DE OPERACIONES -----
select T100_CODE as 'Codigo',T100_BIGALP as 'Descripcion'
from PRODUCCION.MAZP.mazp.BLDT002 with(nolock)   
where T100_LANGUAGE = 'E'  

---- CATALOGO DE PRODUCTOS -----
SELECT DISTINCT key_table [CLAVE PRODUCTO], dta_tblkey  [DESCRIPCION]
from produccion.mazp.mazp.tcdt010 with(nolock)   
where cod_table = '0111'  

---- CATALOGO DE SUB PRODUCTOS -----
SELECT T003_COD_PRODUCT [CLAVE PRODUCTO],T003_CODE [CLAVE SUBPRODUCTO],T003_DES_CODE  [DESCRIPCION] 
FROM PRODUCCION.MAZP.MAZP.WPDT003 WITH(NOLOCK) 

---- CATALOGO DE NACIONALIDAD -----
SELECT KEY_TABLE,DTA_TBLKEY,'' Nacionalidad  
FROM PRODUCCION.MAZP.MAZP.TCDT010 WITH(NOLOCK)  
WHERE COD_TABLE='0412'  


---- CATALOGO DE SUCURSALES -----
SELECT * FROM PRODPLDBAZMEX.DBO.PLD_CAT_SUCURSAL

---- CATALOGO DE CANALES -----
SELECT KEY_TABLE [CLAVE CANAL],DTA_TBLKEY [DESCRIPCION]
FROM PRODUCCION.MAZP.MAZP.TCDT010 WITH(NOLOCK)  
WHERE COD_TABLE='0304'
